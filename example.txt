server {
	server_name join1win.com www.join1win.com;
	charset off;
	index index.php index.html;
	disable_symlinks if_not_owner from=$root_path;
	include /etc/nginx/vhosts-includes/*.conf;
	include /etc/nginx/vhosts-resources/join1win.com/*.conf;
	include /etc/nginx/users-resources/www-root/*.conf;
	access_log /var/www/httpd-logs/join1win.com.access.log;
	error_log /var/www/httpd-logs/join1win.com.error.log notice;
	ssi on;
	set $root_path /var/www/www-root/data/www/join1win.com;
	root $root_path;
	gzip on;
	gzip_comp_level 5;
	gzip_disable "msie6";
	gzip_types text/plain text/css application/json application/x-javascript text/xml application/xml application/xml+rss text/javascript application/javascript image/svg+xml;
	location / {
	try_files $uri $uri/ @modx;
		location ~ [^/]\.ph(p\d*|tml)$ {
			try_files /does_not_exists @php;
		}
		location ~* ^.+\.(jpg|jpeg|gif|png|svg|js|css|mp3|ogg|mpe?g|avi|zip|gz|bz2?|rar|swf|webp|woff|woff2)$ {
        	add_header Cache-Control "public, max-age=3153600, immutable";
        	expires 1y;
        	add_header Last-Modified $date_gmt;
            etag on;
        }
		auth_basic "Access limited by ispmanager";
		auth_basic_user_file /var/www/www-root/data/etc/access.join1win.com.1B2M2Y8A.passwd;
	}
	listen 185.251.25.133:80;
	location @php {
		include /etc/nginx/vhosts-resources/join1win.com/dynamic/*.conf;
		fastcgi_index index.php;
		fastcgi_param PHP_ADMIN_VALUE "sendmail_path = /usr/sbin/sendmail -t -i -f webmaster@join1win.com";
		fastcgi_pass unix:/var/www/php-fpm/52.sock;
		fastcgi_split_path_info ^((?U).+\.ph(?:p\d*|tml))(/?.+)$;
		try_files $uri =404;
		include fastcgi_params;
	}
	location @modx {
        rewrite ^/(.*)$ /index.php?q=$1&$args last;
    }
}
